def calc_R():
    """
    Returns Rijkl
    """
    rnew = np.tile(r,(100,1,1,1))
    rnew = rnew.reshape((100,3,3,3,1000))
    
    rnew_ = np.hstack((r_,np.zeros((100,1))))
    rnew_ = rnew_.reshape((100,3,1,1,1))
    
    return np.linalg.norm(rnew - rnew_ ,axis=1)

def calc_A(l):
    """
    Returns terms of sigma(A)
    """
    Rx,Ry,Rz = r[0,:,:]-r_[l][0], r[1,:,:]-r_[l][1], r[2,:,:]
    R_mag = np.abs(np.sqrt((Rx)**2 + (Ry)**2 + (Rz)**2))
    
    j = complex(0,1)
    k = 0.1
    
    del_phi = np.abs(phi[1]-phi[0])
    dl_x = radius*del_phi*np.sin(phi[l])
    dl_y = radius*del_phi*np.cos(phi[l])
    
    Ax = -1*(np.cos(phi[l])*(np.exp(-j*k*R_mag))*dl_x)/R_mag
    Ay = +1*(np.cos(phi[l])*(np.exp(-j*k*R_mag))*dl_y)/R_mag
    return Ax,Ay

# Bz = cz^b
# log(Bz) = log(c) + b*log(z)
# y = c + m*x which can be written as product of two matrices

y = np.log(Bz)
x = np.log(z)

M = np.c_[np.ones((len(z))),x] # M is the matrix (1 x)
c, m = np.linalg.lstsq(M,y,rcond=None)[0] #least square approach is used to solve the equation
